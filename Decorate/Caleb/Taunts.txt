actor CalebTaunt: CustomInventory
{
    Inventory.MaxAmount 0
    +AUTOACTIVATE
    
    States
    {
       Spawn:
         TNT1 A 0
         stop

       UseStop:
         TNT1 A 0
         stop
    }
}
