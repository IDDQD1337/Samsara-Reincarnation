actor "Slow Fire Protector": InstantSwitchWeapon
{
    Weapon.SelectionOrder 1000
    Weapon.SlotNumber 2
	Weapon.BobStyle Alpha
	Weapon.BobRangeX 0
	Weapon.BobRangeY 0.6
	Weapon.BobSpeed 1.5	
    Inventory.PickUpSound "wolfen/knifeget"
    Obituary "%k \cvENERGIZED!\c- %o \cvwith the Slow Fire Protector.\c-"
    Inventory.PickupMessage "You got a Slow Fire Protector!"
    Weapon.AmmoUse 1
    Weapon.AmmoGive 10
	Weapon.AmmoType "Shell"
    +INVENTORY.UNDROPPABLE
    Weapon.YAdjust 11
	Inventory.RestrictedTo "BlakeStone"		
    States
    {
      Spawn:
        BWP2 A -1
        Loop

      Ready:
        BPI2 A 1 A_WeaponReady
        loop

      Fire:
        TNT1 A 0 A_JumpIfInventory("CoopModeOn",1,"CoopFire")
        BPI2 B 3
        TNT1 A 0 A_PlayWeaponSound("Bpistol2/Fire")
		BPI2 C 3 BRIGHT A_FireCustomMissile("BlakeBullet2",frandom(-2.80,2.80),1,0,1,0,frandom(-2.80,2.80))
      //BPI2 C 3 BRIGHT A_FireBullets(1.8,1.8,1,32,"BlankPuffBlake")
        BPI2 DEA 3
        Goto Ready

      CoopFire:
		BPI2 B 3
        TNT1 A 0 A_PlayWeaponSound("BPistol2/Fire")
        BPI2 C 3 BRIGHT A_FireCustomMissile("BlakeBullet2Coop",frandom(-2.80,2.80),1,0,1,0,frandom(-2.80,2.80))
        BPI2 DEA 3
        Goto Ready
    }
}
