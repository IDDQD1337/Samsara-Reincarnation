ACTOR RoTTM60 : Weapon
{
     Weapon.SlotNumber 4
	 Weapon.SlotPriority 1
     Weapon.SelectionOrder 1800
     Inventory.PickupSound "gen/pickup"
     Inventory.Pickupmessage "You picked up an M60!"
	 Weapon.UpSound "gen/wepswitch"
     Obituary "%k was riddled with bullets by %o's M60."
	 Attacksound "rottm60/shoot"
	 Inventory.RestrictedTo "IPFreeley"
     Tag "M60"
     States 
     { 
     Spawn: 
        PM60 A -1 
        Stop 
     Ready: 
		TNT1 A 0 A_GiveInventory("UsingFirearm", 1)
        RM60 A 1 A_WeaponReady 
        Goto Ready+1
     Deselect: 
        RM60 A 1 A_Lower 
        Loop 
     Select:
        RM60 A 1 A_Raise
        Loop 
     Fire: 
	    TNT1 A 0 A_PlayWeaponSound("rottm60/shoot")
		TNT1 A 0 A_JumpIfInventory("CoopModeOn", 1, "CoopFire")
		TNT1 A 0 A_FireBullets(4.8,3.8,1,15,"RottPuff", FBF_NORANDOM)
	Continue:
        RM60 B 1 Bright A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		TNT1 A 0 A_GunFlash
		RM60 C 1 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		RM60 A 1 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		TNT1 A 0 A_Refire
        RM60 A 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
        Goto Ready+1
	CoopFire:
		TNT1 A 0 A_FireBullets(4.8,3.8,1,15,"RottPuffCoop", FBF_NORANDOM)
		Goto continue
     } 
}
